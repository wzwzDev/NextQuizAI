      </div>
 
      {!loading && rankedFilteredQuizzes.length === 0 && (
        <div className="section-shell mt-8 rounded-2xl p-6 text-sm text-muted-foreground">
          No quizzes found.
        </div>
      )}
    </div>
  );
}
